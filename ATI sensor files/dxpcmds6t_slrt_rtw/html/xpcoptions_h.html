33/* Change all stepsize using the newBaseRateStepSize */



34void dxpcmds6t_ChangeStepSize(real_T newBaseRateStepSize, RT_MODEL_dxpcmds6t_T *



35  const dxpcmds6t_M)



36{



37  real_T ratio = newBaseRateStepSize / 0.001;



38



39  /* update non-zore stepsize of periodic



40   * sample time. Stepsize of asynchronous



41   * sample time is not changed in this function */



42  dxpcmds6t_M->Timing.stepSize0 = dxpcmds6t_M->Timing.stepSize0 * ratio;



43  dxpcmds6t_M->Timing.stepSize1 = dxpcmds6t_M->Timing.stepSize1 * ratio;



44  dxpcmds6t_M->Timing.stepSize2 = dxpcmds6t_M->Timing.stepSize2 * ratio;



45  dxpcmds6t_M->Timing.stepSize3 = dxpcmds6t_M->Timing.stepSize3 * ratio;



46  dxpcmds6t_M->Timing.stepSize = dxpcmds6t_M->Timing.stepSize * ratio;



47}



48



49void XPCCALLCONV changeStepSize(real_T stepSize)



50{



51  /* Change all stepsize using the newBaseRateStepSize */



52  dxpcmds6t_ChangeStepSize(stepSize, dxpcmds6t_M);



53}
